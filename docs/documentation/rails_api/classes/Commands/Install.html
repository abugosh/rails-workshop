393:     def determine_install_method
394:       best = @base_command.environment.best_install_method
395:       @method = :http if best == :http and @method == :export
396:       case
397:       when (best == :http and @method != :http)
398:         msg = "Cannot install using subversion because `svn' cannot be found in your PATH"
399:       when (best == :export and (@method != :export and @method != :http))
400:         msg = "Cannot install using #{@method} because this project is not under subversion."
401:       when (best != :externals and @method == :externals)
402:         msg = "Cannot install using externals because vendor/plugins is not under subversion."
403:       end
404:       if msg
405:         puts msg
406:         exit 1
407:       end
408:       @method

364:       OptionParser.new do |o|
365:         o.set_summary_indent('  ')
366:         o.banner =    "Usage: #{@base_command.script_name} install PLUGIN [PLUGIN [PLUGIN] ...]"
367:         o.define_head "Install one or more plugins."
368:         o.separator   ""
369:         o.separator   "Options:"
370:         o.on(         "-x", "--externals",
371:                       "Use svn:externals to grab the plugin.",
372:                       "Enables plugin updates and plugin versioning.") { |v| @method = :externals }
373:         o.on(         "-o", "--checkout",
374:                       "Use svn checkout to grab the plugin.",
375:                       "Enables updating but does not add a svn:externals entry.") { |v| @method = :checkout }
376:         o.on(         "-e", "--export",
377:                       "Use svn export to grab the plugin.",
378:                       "Exports the plugin, allowing you to check it into your local repository. Does not enable updates or add an svn:externals entry.") { |v| @method = :export }
379:         o.on(         "-q", "--quiet",
380:                       "Suppresses the output from installation.",
381:                       "Ignored if -v is passed (rails plugin -v install ...)") { |v| @options[:quiet] = true }
382:         o.on(         "-r REVISION", "--revision REVISION",
383:                       "Checks out the given revision from subversion or git.",
384:                       "Ignored if subversion/git is not used.") { |v| @options[:revision] = v }
385:         o.on(         "-f", "--force",
386:                       "Reinstalls a plugin if it's already installed.") { |v| @options[:force] = true }
387:         o.separator   ""
388:         o.separator   "You can specify plugin names as given in 'plugin list' output or absolute URLs to "
389:         o.separator   "a plugin repository."
390:       end
391:     end

411:     def parse!(args)
412:       options.parse!(args)
413:       if args.blank?
414:         puts options
415:         exit 1
416:       end
417:       environment = @base_command.environment
418:       install_method = determine_install_method
419:       puts "Plugins will be installed using #{install_method}" if $verbose
420:       args.each do |name|
421:         ::Plugin.find(name).install(install_method, @options)
422:       end
423:     rescue StandardError => e
424:       puts "Plugin not found: #{args.inspect}"
425:       puts e.inspect if $verbose
426:       exit 1
427:     end
